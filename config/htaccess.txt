# Backup Copilot - Security Protection
# Deny all direct access to backup files

# Apache 2.4+
<IfModule mod_authz_core.c>
	Require all denied
</IfModule>

# Apache 2.2
<IfModule !mod_authz_core.c>
	Order deny,allow
	Deny from all
</IfModule>

# Disable directory browsing
<IfModule mod_autoindex.c>
	Options -Indexes
</IfModule>

# Disable script execution
<FilesMatch "\.(php|phtml|php3|php4|php5|pl|py|jsp|asp|html|htm|shtml|sh|cgi)$">
	<IfModule mod_authz_core.c>
		Require all denied
	</IfModule>
	<IfModule !mod_authz_core.c>
		Order deny,allow
		Deny from all
	</IfModule>
</FilesMatch>

# MIME types for backup files
<IfModule mod_mime.c>
	AddType application/zip .zip
	AddType multipart/x-zip .zip
	AddType application/x-zip-compressed .zip
	AddType application/octet-stream .zip
</IfModule>

# Set index file
<IfModule mod_dir.c>
	DirectoryIndex index.php
</IfModule>
